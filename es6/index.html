Autre avantage 

					
function asynchrounFunction(arg, callback){
	callback(JSON.stringify(arg) + " asynchronous result");
};
class MyClass{
	constructor(){
		this.attr = 'My Attribute';
	}
	method(){
		return new Promise(function(res){
			asynchrounFunction(this, function(asynchRes){
				res(asynchRes);
			});
		});
	}
}
var o = new MyClass();
o.method().then(function(myRes){
	console.log(myRes);
})

Promises 

						
let doubleNumberAfterSeconde = function(v){
	return new Promise((resolve, reject)=>{
		setInterval(()=>{
			if(v <= 5){
				resolve(2 * v);
			}else{
				reject("le nombre est tres grand pour etre calculé dans une seconde");
			}
		}, 1000);
	});
};

doubleNumberAfterSeconde(5).then(
	(result) => {
		console.log("seccess : " + result);
	},(error) => {
		console.log("error : " + error);
	}
);
